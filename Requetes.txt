PREFIX k: <http://www.semanticweb.org/evanj/ontologies/2021/4/JeuxVidéos#>

SELECT ?Jeux
	WHERE { ?Jeux rdfs:subClassOf k:Licences}
	ORDERBY ?Jeux 

SELECT ?Genres
	WHERE { ?Genres rdfs:subClassOf k:Genre}
	ORDERBY ?Genres

SELECT ?Jeux
	WHERE {
		?Jeux rdfs:subClassOf k:Licences .
		FILTER(?Jeux = k:God_of_war)
	}

SELECT DISTINCT ?Propriétés
	WHERE
	{
	?Propriétés rdf:type owl:ObjectProperty
	}

SELECT DISTINCT ?Classes
	WHERE
	{
	?Classes rdf:type owl:Class
	} ORDER BY ?Classes
